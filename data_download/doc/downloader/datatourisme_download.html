 53	def _fetch_with_retry(self, url,retries=8, timeout=30):
 54		"""
 55		Sends a GET request with retry logic.
 56
 57		Retries are performed in case of HTTP 429 (rate limiting)
 58		or server errors (5xx), using exponential backoff.
 59
 60		Args:
 61			url (str): URL to request.
 62			retries (int, optional): Maximum number of retry attempts. Defaults to 8.
 63			timeout (int, optional): Request timeout in seconds. Defaults to 30.
 64
 65		Returns:
 66
 67			dict | None: Parsed JSON response if successful, otherwise None.
 68		"""
 69
 70		for attempt in range(retries):  # Boucle de répétition
 71			try:
 72				response = requests.get(url, timeout=timeout)  # Requête pour récupérer les données avec un timeout
 73
 74				if response.status_code == 200:  # Gère les succès de requête
 75					return response.json()  # Renvoie les données JSON converties en dictionnaire Python
 76
 77				if response.status_code == 429:  # Gère l'erreur "trop de requêtes"
 78					wait_time = min(2 ** attempt, 60)  # Calcule le temps d'attente
 79					print(f"[429] Trop de requêtes. Nouvelle tentative dans {wait_time}s...")
 80					time.sleep(wait_time)  # Le programme attend
 81					continue  # Passe à la tentative suivante
 82
 83				if 500 <= response.status_code < 600:  # Gère les erreurs serveur
 84					wait_time = min(2 ** attempt, 60)  # Calcule le temps d'attente
 85					print(f"[{response.status_code}] Erreur serveur. Nouvelle tentative dans {wait_time}s...")
 86					time.sleep(wait_time)  # Le programme attend
 87					continue  # Passe à la tentative suivante
 88
 89				print(f"Erreur HTTP {response.status_code}")  # Gère les autres erreurs HTTP
 90				print(response.text)  # Affiche le détail de l'erreur
 91				return None  # Pas de résultat exploitable
 92
 93			except requests.exceptions.RequestException as e:  # Intercepte les exceptions réseau
 94				wait_time = min(2 ** attempt, 60)  # Calcule le temps d'attente
 95				print(f"Erreur réseau : {e}")
 96				print(f"Nouvelle tentative dans {wait_time}s...")
 97				time.sleep(wait_time)  # Le programme attend
 98
 99		print("Échec après plusieurs tentatives.")
100		return None  # Pas de résultat

215	def extract_data(self):
216		"""
217		Extracts data from the API and stores it in NDJSON files.
218
219		Supports resuming from a checkpoint if available.
220
221		Returns:
222
223			int: Total number of extracted objects.
224		"""
225
226		url = None
227		self._ensure_output_dir()  # Crée le dossier de sortie si nécessaire
228
229		checkpoint = self._load_checkpoint()  # Appelle la fonction pour voir s'il existe un état de reprise
230
231		if checkpoint and ((checkpoint["total_count"]/checkpoint["page_count"]) == self.page_size): # Vérifie si checkpoint est vide ou non
232			print("Checkpoint trouvé. Reprise en cours...")
233			url = checkpoint["next_url"]  # Récupère l'URL stockée dans le checkpoint
234			page_count = checkpoint["page_count"]  # Récupère le nombre de pages déjà traitées
235			file_index = checkpoint["file_index"]  # Récupère le numéro du fichier courant
236			total_count = checkpoint["total_count"]  # Récupère le nombre d'objets déjà traités
237			print(f"Reprise depuis page ~{page_count + 1} | Objets déjà sauvegardés : {total_count}")
238		else:  # Démarre une extraction neuve
239			print("Début de l'extraction...")
240			url = f"{self.url_api}?api_key={self.api_key}&page=1&page_size={self.page_size}&lang=fr"  # Construit l'URL de départ
241			page_count = 0  # Initialise le compteur de pages
242			file_index = 1  # Initialise le numéro du premier fichier
243			total_count = 0  # Initialise le compteur total d'objets
244		while url:  # Tant que url n'est pas None ou vide
245			print(f"Requête : {url}")
246
247			data = self._fetch_with_retry(url)  # Appelle la fonction de requête robuste
248			if not data:  # Vérifie si data est vide
249				print("Arrêt de l'extraction à cause d'une erreur.")
250				self._save_checkpoint(url, total_count, page_count, file_index)  # Sauvegarde le checkpoint
251				return total_count  # Retourne le nombre total d'objets déjà extraits
252
253			objects = data.get("objects", [])  # Récupère la valeur associée à la clé "objects"
254			output_file = self._get_output_filename(file_index)  # Détermine le fichier NDJSON courant
255			self._append_ndjson(objects, output_file)  # Ajoute les objets au fichier courant
256
257			page_count += 1  # Incrémente le compteur de pages
258			total_count += len(objects)  # Met à jour le compteur total d'objets
259			print(f"Page {page_count} récupérée | Total : {total_count} objets")
260
261			next_url = data.get("meta", {}).get("next")  # Cherche l'URL de la page suivante
262
263			if page_count % 5 == 0:  # Vérifie si la page est un multiple de 5
264				print("Sauvegarde intermédiaire...")
265				self._save_checkpoint(next_url, total_count, page_count, file_index)  # Sauvegarde le point de reprise
266
267			if page_count % self.page_file == 0:  # Vérifie s'il faut passer à un nouveau fichier
268				file_index += 1  # Incrémente le numéro de fichier
269				print(f"Changement de fichier : prochain fichier = {self._get_output_filename(file_index)}")
270
271			url = next_url  # Met à jour l'URL suivante
272			time.sleep(self.time_sleep)  # Le programme attend pour éviter de surcharger l'API
273
274		if os.path.exists(self.path_state):  # Vérifie si le fichier temporaire existe
275			os.remove(self.path_state)  # Supprime le fichier temporaire
276			print("Checkpoint supprimé : extraction terminée.")
277
278		return total_count  # Retourne le nombre total d'objets extraits
279
280
281	if __name__ == "__main__":  # Vérifie si le fichier est exécuté directement
282		print("Début extraction...")
283
284		total = extract_data()  # Appelle la fonction qui extrait les données
285
286		print(f"Extraction terminée : {total} objets")

289	def extract_types(self):
290		"""
291		Extracts all "type" values from NDJSON files.
292
293		Iterates over all files in the output directory and collects
294		type fields from each JSON object.
295
296		Returns:
297
298			list: List of extracted types.
299		"""
300
301		all_types = []
302
303		for file_path in Path(self.path_output).glob("*.ndjson"):  # Trouve tous les fichiers .ndjson
304			print(".")
305			with open(file_path, "r", encoding="utf-8") as f:
306				for line_number, line in enumerate(f, start=1):
307					line = line.strip()
308
309					if not line:
310						continue
311
312					try:
313						data = json.loads(line)
314					except json.JSONDecodeError:
315						print(f"Erreur JSON dans {file_path.name}, ligne {line_number}")
316						continue
317
318					types = data.get("type", [])
319
320					if isinstance(types, list):
321						all_types.extend(types)
322
323		return all_types
